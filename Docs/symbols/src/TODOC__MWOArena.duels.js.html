  7 $(document).ready(function () {
  8  
  9 
 10     /**
 11         @namespace MWOArena client api namespace
 12     */
 13      MWOArena={
 14 	 /**
 15          *Clears the MWOArenaMsgBox Div
 16      */
 17           clearMsgBox: function () {
 18              if ($("#MWOArenaMsgBox").length == 0)
 19                  $("#MWOArenaDuels .interface").prepend("<div id='MWOArenaMsgBox'></div>");
 20              $("#MWOArenaMsgBox").empty();
 21          },
 22 		/**
 23          * MWOArena.Duels
 24          *@class MWOArena Duel Management client api
 25 		*/
 26 
 27         Duels : {
 28         
 29             /**
 30               * Submit a duel form to the MWOArenaApis
 31 			*/
 32             addDuel: function () {
 33                 var formData = $('#duel-form').serialize();
 34                 $.support.cors = true;
 35                 $.ajax({
 36                     url: "http://localhost/MWOArena.WebAPIs.Duels/api/duels",
 37                     type: "POST",
 38                     crossDomain: true,
 39                     data: formData,
 40                     dataType: "html",
 41                     success: function (result) {
 42                         MWOArena.clearMsgBox();
 43                         // $('#duel-form').html(result);
 44                         $(".interface").hide();
 45                         $("#MWOArenaMsgBox").html("<b><i>Duel Added</i></b>");
 46                         $("#duel-form")[0].reset(); 
 47                         $("#duel-form").show();
 48                     },
 49                     error: function (jqXHR, tranStatus, errorThrown) {
 50                         $('#duel-form').html("<h3>POST Error!</h3>Browser may not support <a href='http://en.wikipedia.org/wiki/Cross-origin_resource_sharing#Browser_support'>Cross Origin Resource Sharing</a> ");
 51                     }
 52                 });
 53             },
 54 
 55             /**
 56             * Get a list of Duels (filtered by AuthorityName)
 57 			*/
 58             listDuels: function () {
 59                 if ($("#duels-list").length == 0)
 60                     $("#MWOArenaDuels").append("<ul id='duels-list' class='interface list'></ul>");
 61 
 62                 MWOArena.clearMsgBox();
 63 
 64                 $.ajax({
 65                     url: "http://localhost/MWOArena.WebAPIs.Duels/api/duels",
 66                     type: "GET",
 67                     crossDomain: true,
 68                     dataType: "json",
 69                     contentType: "application/json",
 70                     success: function (result) {
 71                         $("#duels-list").hide();
 72                         $.each(result, function () {
 73                             var html = '<center>' + this.MatchGroupName + ' ' + this.DivisionName + '-' + this.ModeName + 's Winner:' + this.WinnerName + ' Loser:' + this.LoserName + '<center>';
 74                             $("#duels-list").append('<li>' + html + '</li>');
 75                         });
 76                         $(".interface").hide();
 77                         $("#duels-list").show();
 78                     },
 79                     error: function (jqXHR, tranStatus, errorThrown) {
 80                         $('#duels-list').html("<h3>POST Error!</h3>Browser may not support <a href='http://en.wikipedia.org/wiki/Cross-origin_resource_sharing#Browser_support'>Cross Origin Resource Sharing</a> ");
 81                     }
 82                 });
 83             }
 84 
 85         
 86 
 87         }
 88     };
 89 
 90 
 91 
 92 
 93 
 94     $("#submit").click(MWOArena.Duels.addDuel);
 95     $("#btnDuels-Add").click(function() {
 96         $(".interface").hide();
 97         $("#duel-form").show();
 98     });
 99     $("#btnDuels-List").click(MWOArena.Duels.listDuels);
100 });
